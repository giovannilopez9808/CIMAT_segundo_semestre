\section*{Problema 03}

\textbf{Compute the gradient of}

\begin{equation*}
    f(\theta) = \frac{1}{2} \sum_{i=1}^n [g(x_i)-g(Ax_i+b)]^2
\end{equation*}

\textbf{with respect to $\theta$, where $\theta=[a_{11},a_{12},a_{21},a_{22},b_1,b_2]^T$, $x_i \in \mathbb{R}^2$ are defined as follows:}

\begin{equation*}
    A = \begin{pmatrix}
        a_{11} & a_{12} \\
        a_{21} & a_{22}
    \end{pmatrix} \qquad
    b= [b_1,b_2]^T
\end{equation*}
