\section*{Problema 02}

\textbf{Suppose $f$ is convex, $\lambda_1 >0$ and $\lambda_2 \leq 0$ with $\lambda_1 + \lambda_2 =1$, and let $x_1, x_2 \in \text{dom } f$. Show that the inequality}

\begin{equation*}
    f(\lambda_1 x_1 + \lambda_2 x_2) \geq \lambda_1 f(x_1) + \lambda_2 f(x_2)
\end{equation*}

\textbf{always holds.}
