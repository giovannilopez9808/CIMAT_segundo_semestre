\section*{Problema 08}


\textbf{Consider the induced matrix norm}

\begin{equation*}
    ||A||_p  = \underset{x\neq 0}{max} \frac{||A_x||_p}{||x||_p}
\end{equation*}

\textbf{where $||\cdot||_p$ is the $l_p$ norm, ie}

\begin{equation*}
    ||x|| = (|x_1|^p+|x_2|^p+\dots+|x_n|^p)^{\frac{1}{p}}
\end{equation*}

\textbf{show that }

\begin{equation*}
    ||AB||_p \le ||A||_p ||B||_p
\end{equation*}
